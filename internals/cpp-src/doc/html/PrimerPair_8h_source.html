44  PrimerPair(const std::string line) {
45  std::vector<std::string> fields;
46  std::string trimmed = boost::trim_copy(line);
47  boost::split(fields, trimmed, boost::is_space(), boost::token_compress_on);
48 
49  if (fields.size() < 4) {
50  throw std::runtime_error("Error: unable to parse incomplete line in primer file.");
51  }
52  try {
53  fw_left = stoi(fields[0]);
54  fw_right = stoi(fields[1]);
55  rv_left = stoi(fields[2]);
56  rv_right = stoi(fields[3]);
57  } catch (std::exception &err) {
58  throw std::runtime_error("Error: line is incorrectly formatted (couldn't parse primer locations).");
59  }
60  }

74 loadPrimerPairs(std::string filename) {
75  std::ifstream file_in(filename, std::ios_base::in | std::ios_base::binary);
76  if (!file_in) {
77  // Do additional checks to see if the file exists or if it's a permissions issue
78  if (!(BF::is_regular_file(filename))) {
79  throw std::runtime_error("ERROR: Input file " + filename + " not found.");
80  }
81  throw std::runtime_error("ERROR: Could not open input file " + filename +
82  " - unknown error.\nCheck file and folder permissions.");
83  }
84  BI::filtering_istream in;
85  // universal newline support filter
86  in.push(BI::newline_filter(BI::newline::posix));
87  in.push(file_in);
88 
89  std::vector<PrimerPair> primer_pairs;
90  std::string line;
91  while (std::getline(in, line)) {
92  std::string trimmed = boost::trim_copy(line);

94  if (trimmed.length()<1 or

96  std::isalpha(trimmed[0])){
