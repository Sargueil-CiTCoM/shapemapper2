31  int charToPhred(char c) {
32  int ascii_value = (int) c;
33  int phred_score = ascii_value - 33;
34  // Do some sanity checks, and throw exception if needed.
35  if (phred_score < 0 or phred_score > 93) {

38  " is out of expected range 0-93. Input quality score lines may have unprintable characters.");
39  }
40  return phred_score;

48  int locateLowQualityWindow(const std::string &phred_scores,
49  unsigned int window_size,
50  unsigned int min_phred) {
51  std::vector<unsigned int> scores(phred_scores.length());
52  for (int i = 0; i < phred_scores.length(); ++i) {
53  try {
54  scores[i] = charToPhred(phred_scores[i]);

57  "ERROR: Phred score string contains whitespace or non-printable characters. Check line endings.");
58  }
59  }
60  int low_qual_index = -1;
61  for (int i = 0; i < scores.size() - window_size; i++) {
62  int sum = 0;
63  for (int j = 0; j < window_size; j++) {
64  sum += scores[i + j];
65  }
66  float mean = (float) sum / (float) window_size;
67  if (mean < min_phred) {
68  low_qual_index = i;
69  break;
70  }
