268  std::vector<Mutation> collapseMutations(const std::vector<Mutation> &mutations,
269  const int max_internal_match,
270  const std::string &local_target_seq,
271  const std::string &local_target_qual,
272  const int left_target_pos) {
273  std::vector<Mutation> collapsed_mutations;
274  std::vector<Mutation> unmerged_mutations; // temp storage for ambiguous mutations (if any),
275  // to avoid merging with other mutations
276  if (mutations.size() > 0) {
277  for (std::vector<Mutation>::const_iterator it = mutations.begin();
278  it != mutations.end();
279  ++it) {
280  Mutation m(*it);
281  // if a mutation is not really a mutation (N match),
282  // it should not be merged with other mutations
283  if (m.tag == "N_match") {
284  unmerged_mutations.push_back(m);
285  } else if (collapsed_mutations.size() > 0 and
286  m.left - (collapsed_mutations.back().right - 1) <= max_internal_match) {
287  // mutation within allowed distance for adjacency
288  std::string seq_sub = local_target_seq.substr(collapsed_mutations.back().right - left_target_pos,
289  m.left - collapsed_mutations.back().right + 1);
290  // ^ pick up nucs between mutations
291  if (seq_sub.find('_') != std::string::npos) {
292  // don't collapse mutations across the gap between R1 and R2 if present
293  // (indicated by '_' chars in sequence)
294  collapsed_mutations.push_back(m);
295  } else {
296  collapsed_mutations.back().seq += seq_sub;
297  collapsed_mutations.back().qual += local_target_qual.substr(
298  collapsed_mutations.back().right - left_target_pos,
299  m.left - collapsed_mutations.back().right + 1);
300  collapsed_mutations.back().right = m.right;
301  collapsed_mutations.back().seq += m.seq;
302  collapsed_mutations.back().qual += m.qual;
303  collapsed_mutations.back().tag = "";
304  if (m.ambig) {
305  collapsed_mutations.back().ambig = true;
306  }
307  }
308  } else {
309  // non-adjacent mutation
310  collapsed_mutations.push_back(m);
311  }
312  }
313  // iteratively strip any matches from mutation ends (these can appear as a result of
314  // shifted indels). For left-aligned ambig gaps, this only affects the 5-prime end of
315  // a mutation, so shouldn't affect reactivity profiles. Adjusting it anyway for
316  // consistency between aligners.
317  if (collapsed_mutations.size() > 0) {
318  for (std::vector<Mutation>::iterator it = collapsed_mutations.begin();
319  it != collapsed_mutations.end();
320  ++it) {
321  // from left end
322  int new_left = (*it).left;
323  std::string new_seq = (*it).seq;
324  std::string new_qual = (*it).qual;
325  try {
326  for (int i = 0; i < (*it).seq.length(); ++i) {
327  std::string c = (*it).seq.substr(i, 1);
328  std::string q = (*it).qual.substr(i, 1);
329  if ((*it).left + 1 + i >=
330  (*it).right) { break; } // don't go outside of region mutation spans
331  int p = (*it).left + 1 + i - left_target_pos;
332  if (p < 0) { break; }
333  std::string r = local_target_seq.substr(p, 1);
334  if (c == r) {
335  ++new_left;
336  new_seq = new_seq.substr(1);
337  new_qual = new_qual.substr(1);
338  } else {
339  break;
340  }
341  }
342  }
343  catch (std::out_of_range &exception) {
344  break;
345  }
346  (*it).left = new_left;
347  (*it).seq = new_seq;
348  (*it).qual = new_qual;
349 
350  // from right end
351  int new_right = (*it).right;
352  new_seq = (*it).seq;
353  new_qual = (*it).qual;
354  try {
355  for (int i = (*it).seq.length() - 1; i >= 0; --i) {
356  std::string c = (*it).seq.substr(i, 1);
357  std::string q = (*it).qual.substr(i, 1);
358  int d = (*it).seq.length() - i;
359  if ((*it).right - d <= (*it).left) { break; } // don't go outside of region mutation spans
360  int p = (*it).right - d - left_target_pos;
361  if (p < 0) { break; }
362  std::string r = local_target_seq.substr(p, 1);
363  if (c == r) {
364  --new_right;
365  new_seq = new_seq.substr(0, i);
366  new_qual = new_qual.substr(0, i);
367  } else {
368  break;
369  }
370  }
371  }
372  catch (std::out_of_range &exception) {
373  break;
374  }
375  (*it).right = new_right;
376  (*it).seq = new_seq;
377  (*it).qual = new_qual;
378  }
379  }
380 
381  // put unmerged (mismatched Ns) mutations back into list
382  collapsed_mutations.insert(collapsed_mutations.end(),
383  unmerged_mutations.begin(),
384  unmerged_mutations.end());
385  // make sure unmerged are back in order
386  std::sort(collapsed_mutations.begin(), collapsed_mutations.end());
387  }
388  return collapsed_mutations;
389  }

767  Read
768  mergeMatePairs(std::vector<Read> reads) {

771  std::string qual;

773  std::vector<bool> mapped_depth;
774  int left;
775  int right;
776  int length;

778  Read r1(reads[0]);
779  Read r2(reads[1]);
780 
781  // seq can be merged no problem, there should be no conflicts
782  // for qual, just use the higher score in overlap
783  // conflicting mutations are trickier
784 
785  left = std::min(r1.left, r2.left);
786  right = std::max(r1.left + r1.seq.size() - 1,
787  r2.left + r2.seq.size() - 1);
788  length = right - left + 1;
789  seq.resize(length);
790  qual.resize(length);
791 
792  Read simple_merged = mergeMatePairsSimple(reads);
793  mapped_depth = simple_merged.mapped_depth;
794 
795  char one, two;
796  for (int i = 0; i < length; i++) {
797  seq.at(i) = '_';
798  qual.at(i) = '~';
799  try {
800  one = r1.seq.at(i - r1.left + left);
801  seq.at(i) = one;
802  } catch (const std::out_of_range &exception) { }
803  try {
804  two = r2.seq.at(i - r2.left + left);
805  seq.at(i) = two;
806  } catch (const std::out_of_range &exception) { }
807 
808  one = '~';
809  two = '~';
810  try {
811  one = r1.qual.at(i - r1.left + left);
812  } catch (const std::out_of_range &exception) { }
813  try {
814  two = r2.qual.at(i - r2.left + left);
815  } catch (const std::out_of_range &exception) { }
816  if (one != '~' and two != '~') {
817  if ((int) one >= (int) two) {
818  qual.at(i) = one;
819  } else {
820  qual.at(i) = two;
821  }
822  }
823  else if (two == '~' and one != '~') {
824  qual.at(i) = one;
825  }
826  else if (one == '~' and two != '~') {
827  qual.at(i) = two;
828  }
829  }
830 
831 
832  // now need to merge (possibly conflicting) mutations
833  // index R1 mutations by left-most nuc
834  std::vector<std::vector<Mutation>> indexed_r1_muts, indexed_r2_muts;
835  indexed_r1_muts.resize(length);
836  indexed_r2_muts.resize(length);
837  for (auto &m : r1.mutations) {
838  Mutation mc(m);
839  indexed_r1_muts.at(m.left - left).push_back(mc);
840  }
841  // index R2 mutations by left-most nuc
842  for (auto &m : r2.mutations) {
843  Mutation mc(m);
844  indexed_r2_muts.at(m.left - left).push_back(mc);
845  }
846  // group all overlapping mutations
847  std::vector<MutationGroup> mutation_groups;
848  MutationGroup mutation_group;
849  //Mutation m1, m2;
850  std::vector<Mutation> v1, v2;
851  for (int i = 0; i < length; i++) {
852  v1 = indexed_r1_muts[i];
853  v2 = indexed_r2_muts[i];
854  for (auto &m1 : v1) {
855  if (mutation_group.r1_mutations.size() == 0 and
856  mutation_group.r2_mutations.size() == 0) {
857  mutation_group.r1_mutations.push_back(m1);
858  mutation_group.left = m1.left;
859  mutation_group.right = m1.right;
860  } else {
861  // check for overlap
862  if (m1.left < mutation_group.right) {
863  mutation_group.r1_mutations.push_back(m1);
864  mutation_group.right = std::max(m1.right, mutation_group.right);
865  } else {
866  mutation_groups.push_back(mutation_group);
867  mutation_group = MutationGroup();
868  mutation_group.r1_mutations.push_back(m1);
869  mutation_group.left = m1.left;
870  mutation_group.right = m1.right;
871  }
872  }
873  }
874  for (auto &m2 : v2) {
875  if (mutation_group.r1_mutations.size() == 0 and
876  mutation_group.r2_mutations.size() == 0) {
877  mutation_group.r2_mutations.push_back(m2);
878  mutation_group.left = m2.left;
879  mutation_group.right = m2.right;
880  } else {
881  // check for overlap
882  if (m2.left < mutation_group.right) {
883  mutation_group.r2_mutations.push_back(m2);
884  mutation_group.right = std::max(m2.right, mutation_group.right);
885  } else {
886  mutation_groups.push_back(mutation_group);
887  mutation_group = MutationGroup();
888  mutation_group.r2_mutations.push_back(m2);
889  mutation_group.left = m2.left;
890  mutation_group.right = m2.right;
891  }
892  }
893  }
894  }
895  // handle final group
896  if (mutation_group.r1_mutations.size() > 0 or
897  mutation_group.r2_mutations.size() > 0) {
898  mutation_groups.push_back(mutation_group);
899  }
900 
901  r1.depth.resize(r1.seq.size(), 1);
902  r2.depth.resize(r2.seq.size(), 1);
903 
904  for (auto &mg : mutation_groups) {
905 
906  // resolve overlaps by choosing the higher quality read mutations
907  float mean_qual_r1, mean_qual_r2;
908  int num = 0;
909  int denom = 0;
910  if (mg.r1_mutations.size() > 0) {
911  for (std::vector<Mutation>::const_iterator m = mg.r1_mutations.begin();
912  m != mg.r1_mutations.end();
913  ++m) {
914  for (int i = 0; i < (*m).qual.size(); i++) {
915  num += (int) (*m).qual.at(i);
916  denom++;
917  }
918  // also consider basecalls neighboring the mutation
919  try {
920  num += (int) r1.qual.at((*m).left - r1.left);
921  denom++;
922  } catch (const std::out_of_range &exception) { }
923  try {
924  num += (int) r1.qual.at((*m).right - r1.left);
925  denom++;
926  } catch (const std::out_of_range &exception) { }
927  }
928  } else {
929  // mutations in r2, but not in r1.
930  // consider basecalls in r1 over same region as mutation group in r2, if r1 covers
931  // left and right
932  int lindex = mg.left - r1.left;
933  int rindex = mg.right - r1.left;
934  if (lindex >= 0 and lindex < r1.qual.size() and
935  rindex >= 0 and rindex < r1.qual.size()) {
936  for (int p = mg.left; p <= mg.right; p++) {
937  num += (int) r1.qual.at(p - r1.left);
938  denom++;
939  }
940  }
941  }
942  if (denom > 0.0) {
943  mean_qual_r1 = float(num) / float(denom);
944  } else {
945  mean_qual_r1 = 0.0;
946  }
947  num = 0;
948  denom = 0;
949  if (mg.r2_mutations.size() > 0) {
950  for (std::vector<Mutation>::const_iterator m = mg.r2_mutations.begin();
951  m != mg.r2_mutations.end();
952  ++m) {
953  for (int i = 0; i < (*m).qual.size(); i++) {
954  num += (int) (*m).qual.at(i);
955  denom++;
956  }
957  // also consider basecalls neighboring the mutation
958  try {
959  num += (int) r2.qual.at((*m).left - r2.left);
960  denom++;
961  } catch (const std::out_of_range &exception) { }
962  try {
963  num += (int) r2.qual.at((*m).right - r2.left);
964  denom++;
965  } catch (const std::out_of_range &exception) { }
966  }
967  } else {
968  // mutations in r1, but not in r2.
969  // consider basecalls in r2 over same region as mutation group in r1, if r2 covers
970  // left and right
971  int lindex = mg.left - r2.left;
972  int rindex = mg.right - r2.left;
973  if (lindex >= 0 and lindex < r2.qual.size() and
974  rindex >= 0 and rindex < r2.qual.size()) {
975  for (int p = mg.left; p <= mg.right; p++) {
976  num += (int) r2.qual.at(p - r2.left);
977  denom++;
978  }
979  }
980  }
981  if (denom > 0.0) {
982  mean_qual_r2 = float(num) / float(denom);
983  } else {
984  mean_qual_r2 = 0.0;
985  }
986 
987  int selected_read = READ1;
988  if (mean_qual_r2 > mean_qual_r1) {
989  selected_read = READ2;
990  }
991  if (selected_read == READ1) {
992  for (int i = 0; i < mg.r1_mutations.size(); i++) {
993  mutations.push_back(mg.r1_mutations[i]);
994  }
995  // remove excluded mutations from r2 effective depth
996  if (mg.r2_mutations.size() > 0) {
997  for (int n = minLeft(mg.r2_mutations) + 1;
998  n < maxRight(mg.r2_mutations);
999  n++) {
1000  try {
1001  r2.depth.at(n - r2.left) = 0;
1002  } catch (std::out_of_range &ex) { }
1003  }
1004  }
1005 
1006  } else {
1007  for (int i = 0; i < mg.r2_mutations.size(); i++) {
1008  mutations.push_back(mg.r2_mutations[i]);
1009  }
1010  // remove excluded mutations from r1 effective depth
1011  if (mg.r1_mutations.size() > 0) {
1012  for (int n = minLeft(mg.r1_mutations) + 1;
1013  n < maxRight(mg.r1_mutations);
1014  n++) {
1015  try {
1016  r1.depth.at(n - r1.left) = 0;
1017  } catch (std::out_of_range &ex) { }
1018  }
1019  }
1020 
1021  }
1022 
1023 
1024  }
1025 
1026  // combine effective depths from r1 and r2
1027  depth.resize(seq.size(), 0);
1028  for (int i = 0; i < depth.size(); i++) {
1029  bool d1, d2;
1030  d1 = 0;
1031  d2 = 0;
1032  int loc = left + i;
1033  int r1_index = loc - r1.left;
1034  int r2_index = loc - r2.left;
1035  try {
1036  d1 = r1.depth.at(r1_index);
1037  } catch (std::out_of_range &ex) { }
1038  try {
1039  d2 = r2.depth.at(r2_index);
1040  } catch (std::out_of_range &ex) { }
1041  bool m = 0;
1042  if (d1 or d2) { m = 1; }
1043  depth[i] = m;
1044  }
1045 
1046 
1047  return Read(left,
1048  right,
1049  seq)
1050  .setReadType(PAIRED)
1051  .setStrand(FORWARD)
1052  .setId(r1.id)
1053  .setMappingCategory(r1.mapping_category)
1054  .setMappedDepth(mapped_depth)
1055  .setPrimerPair(r1.primer_pair)
1056  .setQual(qual)
1057  .setMutations(mutations)
1058  .setDepth(depth);
1059  }
