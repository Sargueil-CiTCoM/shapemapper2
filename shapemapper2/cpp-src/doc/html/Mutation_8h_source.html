204  std::string classify(const std::string &local_target,
205  const int target_pos) const {
206  int d = right - left - 1;
207  std::string s;
208  try {
209  if (d == 1 and seq.length() == 0) {
210  // deletion of a single nucleotide
211  int i = left + 1 - target_pos;
212  s = local_target.substr(left + 1 - target_pos, 1) + '-';
213  } else if (d == 0 and seq.length() == 1) {
214  // insertion of a single nucleotide
215  s = '-' + seq;
216  } else if (d == 1 and seq.length() == 1) {
217  // single-nucleotide mismatch
218  if (seq == "N") {
219  s = "N_match";
220  } else {
221  s = local_target[left + 1 - target_pos] + seq;
222  }
223  } else if (d > 1 and seq.length() == 0) {
224  s = "multinuc_deletion";
225  } else if (d == 0 and seq.length() > 1) {
226  s = "multinuc_insertion";
227  } else if (d == seq.length()) {
228  s = "multinuc_mismatch";
229  } else if (seq.length() < d) {
230  s = "complex_deletion";
231  } else if (seq.length() > d) {
232  s = "complex_insertion";
233  } else {
234  throw std::runtime_error("Error: Unknown error. Unable to classify mutation.");
235  }
236  } catch (std::out_of_range &err) {
237  throw std::runtime_error(
238  "Error: Unable to classify mutation. Mutation location falls outside local target sequence.");
239  }
240  return s;
241  }

315  }
316  for (int i = start_index; i <= fields.size() - 5; i += 5) {
317  try {
318  Mutation m(stoi(fields[i]),
319  stoi(fields[i + 1]),
320  fields[i + 2].substr(1, fields[i + 2].length() - 2),
321  fields[i + 3].substr(1, fields[i + 3].length() - 2),
322  fields[i + 4].substr(1, fields[i + 4].length() - 2)
323  );
324  mutations.push_back(m);
325  } catch (std::exception &err) {
326  throw std::runtime_error(
327  "Error: line is incorrectly formatted (couldn't read mutation left or right bounds).");
328  }
329  }
330  return mutations;
331 }
