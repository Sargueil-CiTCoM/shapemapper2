182 Read::Read(const std::string &serialized){
183  std::vector<std::string> fields;
184  std::string trimmed = boost::trim_right_copy_if(serialized, boost::is_any_of("\n\r"));
185  boost::split(fields, trimmed, boost::is_any_of("\t"));
186  read_type = UNSPECIFIED_READ_TYPE;
187  primer_pair = NO_ASSOCIATED_PRIMER_PAIR;
188  mapping_category = INCLUDED;
189  if (fields.at(1) != "") {
190  read_type = util::indexOf(read_types, fields.at(1));
191  }
192  left = stoi(fields.at(2));
193  right = stoi(fields.at(3));
194  strand = UNSPECIFIED_STRAND;
195  if (fields.at(4) == "+"){ strand = FORWARD; }
196  if (fields.at(4) == "-"){ strand = REVERSE; }
197  mapping_category = util::indexOf(mapping_categories, fields.at(5));
198  primer_pair = std::stoi(fields.at(6));
199  seq = fields.at(7);
200  qual = fields.at(8);
201  try {
202  mapped_depth = util::stringToBoolVect(fields.at(9));
203  depth = util::stringToBoolVect(fields.at(10));
204  count = util::stringToBoolVect(fields.at(11));
205  mutations = stringToMutationVect(fields.at(12));
206  } catch (const std::out_of_range& ex) { }
207 }

318 Read::toString() const {
319  std::string s = "";
320  s += "[read]\t";
321  s += read_types[read_type] + '\t';
322  s += std::to_string(left) + '\t';
323  s += std::to_string(right) + '\t';
324  std::string strand_str = "N/A";
325  if (strand == FORWARD) { strand_str = "+"; }
326  if (strand == REVERSE) { strand_str = "-"; }
327  s += strand_str + '\t';
328  s += mapping_categories.at(mapping_category) + '\t';
329  s += std::to_string(primer_pair) + '\t';
330  s += seq + '\t';
331  s += qual + '\t';
332  s += util::toString(mapped_depth) + '\t';
333  s += util::toString(depth) + '\t';
334  s += util::toString(count) + '\t';
335  s += mutation::toString(mutations);
336  s += "\n";
337  return s;

473 Read
474 mergeMatePairsSimple(const std::vector<Read> &reads,
475  const bool debug = false){
476  /*
477  * Simple merge of two mate paired reads, without considering sequence or mutations.
478  * Used to allow read depth counting for off-target and low mapping quality reads.
479  */
480  Read simple_merged{}; // explicitly call default constructor
481  std::vector<bool> mapped_depth;
482 
483  int fw_read_index = R1;
484  int rv_read_index = R2;
485  if (reads[R1].strand == REVERSE and reads[R2].strand == FORWARD) {
486  fw_read_index = R2;
487  rv_read_index = R1;
488  }
489 
490  int left = std::min(reads.at(fw_read_index).left, reads.at(rv_read_index).left);
491  int right = std::max(reads.at(rv_read_index).right, reads.at(fw_read_index).right);
492  mapped_depth.resize(right - left + 1, 0);
493  if (debug) {
494  std::cout << "mapped_depth.size() after resize: " << mapped_depth.size() << "\n" << std::flush;
495  }
496  int l1 = reads.at(fw_read_index).right - reads.at(fw_read_index).left + 1;
497  int l2 = reads.at(rv_read_index).right - reads.at(rv_read_index).left + 1;
498  std::fill(mapped_depth.begin(), mapped_depth.begin() + l1, 1);
499  if (debug) {
500  std::cout << "filled fw read depth\n" << std::flush;
501  }
502  std::fill(mapped_depth.end() - l2, mapped_depth.end(), 1);
503  if (debug) {
504  std::cout << "filled fw read depth\n" << std::flush;
505  }
506  simple_merged.setReadType(PAIRED)
507  .setId(reads[0].id)
508  .setLeft(left)
509  .setRight(right)
510  .setMappedDepth(mapped_depth);
511  return simple_merged;
512 }
